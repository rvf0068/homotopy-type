def _read_betti_numbers(bettis):
    betti = "\\("
    for index in range(len(bettis)):
        if bettis[index] != 0:
            if bettis[index] == 1:
                betti = betti + f"S^{ {index} }"
            else:
                betti = betti + f"\\vee_{ {bettis[index]} }S^{ {index} }"
            betti = betti + "\\vee "        
    return betti[:-5] + "\\)"
